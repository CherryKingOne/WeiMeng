### 二、 核心代码实现

#### 1. API 路由层 (`src/modules/agent/api/router.py`)
确保最终挂载到 `v1/router.py` 中的前缀是 `/agent`。

```python
from fastapi import APIRouter, Depends
from fastapi.responses import StreamingResponse
from ...shared.common.response import SuccessResponse
from ..application.dto.agent_chat_dto import AgentChatRequest
from ..application.services.agent_app_service import AgentAppService
from .dependencies import get_agent_app_service

router = APIRouter(tags=["Agent"])

@router.post("/chat", summary="Agent 会话接口")
async def agent_chat(
    request: AgentChatRequest,
    agent_service: AgentAppService = Depends(get_agent_app_service)
):
    if request.stream:
        # 流式输出
        return StreamingResponse(
            agent_service.chat_stream(request),
            media_type="text/event-stream"
        )
    else:
        # 非流式输出
        result = await agent_service.chat_sync(request)
        return SuccessResponse(data={"session_id": request.session_id, "reply": result})
```

#### 2. DTO 定义 (`src/modules/agent/application/dto/agent_chat_dto.py`)

```python
from pydantic import BaseModel, Field
from typing import Optional
import uuid

class AgentChatRequest(BaseModel):
    session_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    message: str = Field(..., description="用户输入的消息")
    provider: str = Field(..., description="模型供应商 (例如: openai-compatible)")
    model_name: str = Field(..., description="模型名称 (例如: deepseek-chat)")
    stream: bool = Field(default=False, description="是否流式输出")
    thinking: bool = Field(default=False, description="是否开启供应商思考机制")
```

#### 3. ⭐️ 思考机制适配层 (`src/modules/agent/infrastructure/adapters/thinking_adapter.py`)
解决不同供应商“思考模式”调用差异的核心痛点。

```python
from langchain_core.language_models.chat_models import BaseChatModel

class ThinkingAdapter:
    """模型思考机制适配层"""
    
    @staticmethod
    def apply_thinking(llm: BaseChatModel, provider: str, model_name: str, thinking: bool) -> BaseChatModel:
        """
        根据不同的 provider 和 thinking 开关，对 LangChain 的 LLM 实例进行动态调整。
        """
        if not thinking:
            return llm

        # --- 针对不同供应商开启思考模式 ---
        if provider == "openai-compatible" or provider == "deepseek":
            # DeepSeek 的思考模式通常是将模型切换为 reasoner
            if "deepseek" in model_name.lower():
                llm.model_name = "deepseek-reasoner"
            # 兼容 OpenAI o1/o3 协议的推理力度参数
            if hasattr(llm, "model_kwargs"):
                llm.model_kwargs = {**llm.model_kwargs, "reasoning_effort": "high"}
                
        elif provider == "qwen":
            # 假设 Qwen 开启思考需要特定的参数 qwen_thinking
            if hasattr(llm, "model_kwargs"):
                llm.model_kwargs = {**llm.model_kwargs, "enable_search": True, "reasoning": True}
                
        elif provider == "volcengine":
            # 火山引擎适配
            pass
            
        # 其他供应商按需扩展...
        
        return llm
```

#### 6. LangGraph ReAct Agent 定义 (`src/modules/agent/infrastructure/graph/react_agent.py`)

```python
from langgraph.prebuilt import create_react_agent
from langgraph.checkpoint.memory import MemorySaver
from langchain_core.language_models.chat_models import BaseChatModel
from ..tools.registry import AGENT_TOOLS
from ..workspace.prompts import AGENT_SYSTEM_PROMPT

def build_react_agent(llm: BaseChatModel, memory_saver: MemorySaver):
    """
    构建并返回一个 LangGraph ReAct Agent 执行器
    """
    # 动态组装 Prompt
    tool_desc = ", ".join([t.name for t in AGENT_TOOLS])
    system_message = AGENT_SYSTEM_PROMPT.format(tool_descriptions=tool_desc)
    
    # 使用 LangGraph 内置的 ReAct 范式创建 Agent
    agent_executor = create_react_agent(
        model=llm,
        tools=AGENT_TOOLS,
        checkpointer=memory_saver, # 接入 memory 文件夹中的记忆管理
        state_modifier=system_message
    )
    
    return agent_executor
```

#### 7. App Service 应用服务 (`src/modules/agent/application/services/agent_app_service.py`)

```python
import json
from typing import AsyncGenerator
from langgraph.checkpoint.memory import MemorySaver
from langchain_core.messages import HumanMessage

from ..dto.agent_chat_dto import AgentChatRequest
from ...infrastructure.adapters.thinking_adapter import ThinkingAdapter
from ...infrastructure.graph.react_agent import build_react_agent
# 假设你现有的 providers 模块有获取模型实例的方法
# from src.modules.providers.application.services.chat_service import get_langchain_llm

class AgentAppService:
    def __init__(self):
        # 实际项目中，这里应该使用依赖注入的 RedisSaver 或 SQLiteSaver 存放至 memory 文件夹
        self.memory = MemorySaver() 

    async def _prepare_agent(self, request: AgentChatRequest):
        """准备和初始化 Agent 执行器"""
        
        # 1. 调用现有的 providers 模块获取底层的 Langchain ChatModel 实例
        # 伪代码： llm = await get_langchain_llm(request.provider, request.model_name)
        from langchain_openai import ChatOpenAI # 仅作占位示意
        base_llm = ChatOpenAI(model=request.model_name) 
        
        # 2. 注入 Thinking 思考适配层（核心！）
        llm_with_thinking = ThinkingAdapter.apply_thinking(
            llm=base_llm,
            provider=request.provider,
            model_name=request.model_name,
            thinking=request.thinking
        )
        
        # 3. 构建 LangGraph ReAct Agent
        agent = build_react_agent(llm_with_thinking, self.memory)
        return agent

    async def chat_sync(self, request: AgentChatRequest) -> str:
        """非流式对话"""
        agent = await self._prepare_agent(request)
        config = {"configurable": {"thread_id": request.session_id}}
        
        inputs = {"messages": [HumanMessage(content=request.message)]}
        response = await agent.ainvoke(inputs, config=config)
        
        # 提取最后一条 AI 回复
        return response["messages"][-1].content

    async def chat_stream(self, request: AgentChatRequest) -> AsyncGenerator[str, None]:
        """流式对话 (SSE 格式或 Chunk 格式)"""
        agent = await self._prepare_agent(request)
        config = {"configurable": {"thread_id": request.session_id}}
        inputs = {"messages": [HumanMessage(content=request.message)]}
        
        # 使用 LangGraph 的 astream_events 来精细捕获流式输出（支持工具调用和模型思考过程输出）
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            # 捕获大模型生成的文本块（含思考过程或最终回答）
            if kind == "on_chat_model_stream":
                chunk = event["data"]["chunk"]
                
                # 如果模型供应商返回了特殊的 reasoning_content (如 Deepseek)
                if hasattr(chunk, "additional_kwargs") and "reasoning_content" in chunk.additional_kwargs:
                    reasoning_text = chunk.additional_kwargs["reasoning_content"]
                    if reasoning_text:
                         yield f"data: {json.dumps({'type': 'thinking', 'content': reasoning_text})}\n\n"
                
                # 常规内容
                if chunk.content:
                    yield f"data: {json.dumps({'type': 'message', 'content': chunk.content})}\n\n"
            
            # 捕获工具调用开始
            elif kind == "on_tool_start":
                tool_name = event["name"]
                yield f"data: {json.dumps({'type': 'tool_call', 'content': f'正在使用工具: {tool_name} ...'})}\n\n"
                
        yield "data: [DONE]\n\n"
```

### 三、 集成与注册说明

1. **注册路由**：
   在现有的 `src/api/v1/router.py` 文件中，导入并注册新模块的路由：
   ```python
   from src.modules.agent.api.router import router as agent_router
   api_v1_router.include_router(agent_router, prefix="/agent")
   ```

2. **对接已有的 `providers` 模块**：
   在 `AgentAppService._prepare_agent` 方法中，你需要对接现有系统 `providers` 目录里的 `ChatService` 或 `Factories`。现有的 `providers` 既然已经封装好了，只需让它返回一个满足 `BaseChatModel`（Langchain 标准）的实例，然后交给 `ThinkingAdapter` 加工即可。

3. **Thinking 字段的灵活控制**：
   你的要求被完美实现在了 `ThinkingAdapter` 中。无论外部请求怎样传入 `thinking: true`，Adapter 都会将它翻译为当前 `provider` 对应的特殊参数（如改模型名为 `deepseek-reasoner`，或增加 `reasoning_effort="high"` 参数）。这使得前端和 API 调用方完全解耦。
