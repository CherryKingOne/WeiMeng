#### 1. API 路由层 (`src/modules/agent/api/router.py`)
确保最终挂载到 `v1/router.py` 中的前缀是 `/agent`。

```python
from fastapi import APIRouter, Depends
from fastapi.responses import StreamingResponse
from ...shared.common.response import SuccessResponse
from ..application.dto.agent_chat_dto import AgentChatRequest
from ..application.services.agent_app_service import AgentAppService
from .dependencies import get_agent_app_service

router = APIRouter(tags=["Agent"])

@router.post("/chat", summary="Agent 会话接口")
async def agent_chat(
    request: AgentChatRequest,
    agent_service: AgentAppService = Depends(get_agent_app_service)
):
    if request.stream:
        # 流式输出
        return StreamingResponse(
            agent_service.chat_stream(request),
            media_type="text/event-stream"
        )
    else:
        # 非流式输出
        result = await agent_service.chat_sync(request)
        return SuccessResponse(data={"session_id": request.session_id, "reply": result})
```

#### 3. ⭐️ 思考机制适配层 (`src/modules/agent/infrastructure/adapters/thinking_adapter.py`)
解决不同供应商“思考模式”调用差异的核心痛点。

```python
from langchain_core.language_models.chat_models import BaseChatModel

class ThinkingAdapter:
    """模型思考机制适配层"""
    
    @staticmethod
    def apply_thinking(llm: BaseChatModel, provider: str, model_name: str, thinking: bool) -> BaseChatModel:
        """
        根据不同的 provider 和 thinking 开关，对 LangChain 的 LLM 实例进行动态调整。
        """
        if not thinking:
            return llm

        # --- 针对不同供应商开启思考模式 ---
        if provider == "openai-compatible" or provider == "deepseek":
            # DeepSeek 的思考模式通常是将模型切换为 reasoner
            if "deepseek" in model_name.lower():
                llm.model_name = "deepseek-reasoner"
            # 兼容 OpenAI o1/o3 协议的推理力度参数
            if hasattr(llm, "model_kwargs"):
                llm.model_kwargs = {**llm.model_kwargs, "reasoning_effort": "high"}
                
        elif provider == "qwen":
            # 假设 Qwen 开启思考需要特定的参数 qwen_thinking
            if hasattr(llm, "model_kwargs"):
                llm.model_kwargs = {**llm.model_kwargs, "enable_search": True, "reasoning": True}
                
        elif provider == "volcengine":
            # 火山引擎适配
            pass
            
        # 其他供应商按需扩展...
        
        return llm
```

from langgraph.checkpoint.postgres import PostgresSaver

class PostgresSaverManager:
    """
    PostgreSQL Checkpointer 管理器
    基于 langgraph-checkpoint-postgres 实现跨运行的持久化和历史回溯
    """
    
    @staticmethod
    async def create_and_setup(db_uri: str) -> PostgresSaver:
        """
        创建并初始化 PostgreSQL Checkpointer
        
        Args:
            db_uri: PostgreSQL 连接字符串，例如：postgresql://user:password@localhost:5432/my_database
            
        Returns:
            已初始化的 PostgresSaver 实例
        """
        # 使用 AsyncConnection 来管理连接池
        conn = await AsyncConnection.connect(db_uri)
        
        # 初始化 PostgresSaver
        checkpointer = PostgresSaver(conn)
        
        # 确保数据库表已创建（必须步骤）
        await checkpointer.setup()
        
        return checkpointer

#### 6. ⭐️ 技能加载器 (`src/modules/agent/infrastructure/loading/skills_loader.py`)

```python
import importlib
from pathlib import Path
from typing import List, Dict, Any, Optional
from langchain_core.tools import BaseTool, tool
from langchain.agents import create_agent

class SkillsLoader:
    """
    LangChain Skills 加载器
    实现"按需加载"（Progressive Disclosure）机制
    
    核心理念：
    - 智能体不预先加载所有技能
    - 通过专门的 load_skill 工具在需要时动态获取
    - 支持层级化技能和动态工具注册
    """
    
    def __init__(self, skills_dir: str = None):
        """
        初始化技能加载器
        
        Args:
            skills_dir: skills 文件夹路径，默认为当前模块的 skills 目录
        """
        if skills_dir is None:
            # 默认指向 infrastructure/skills 目录
            self.skills_dir = Path(__file__).parent.parent / "skills"
        else:
            self.skills_dir = Path(skills_dir)
        
        self._loaded_skills: Dict[str, Dict[str, Any]] = {}
        self._available_skills: List[str] = []
        self._scan_available_skills()
    
    def _scan_available_skills(self) -> List[str]:
        """
        扫描 skills 文件夹中所有可用的技能包
        
        Returns:
            技能包名称列表
        """
        self._available_skills = []
        for skill_folder in self.skills_dir.iterdir():
            if skill_folder.is_dir() and not skill_folder.name.startswith('_'):
                # 检查是否包含 SKILL.md 文件
                if (skill_folder / "SKILL.md").exists():
                    self._available_skills.append(skill_folder.name)
        return self._available_skills
    
    def get_available_skills_list(self) -> str:
        """
        获取可用技能列表的描述字符串
        
        Returns:
            格式化后的技能列表描述
        """
        if not self._available_skills:
            return "暂无可用技能"
        
        skills_desc = "可用技能列表:\n"
        for skill_name in self._available_skills:
            skills_desc += f"- {skill_name}\n"
        return skills_desc
    
    def load_skill(self, skill_name: str) -> Dict[str, Any]:
        """
        按需加载指定技能包
        
        Args:
            skill_name: 技能包名称（如 'search_skill'）
            
        Returns:
            包含技能信息的字典：
            {
                'name': str,          # 技能名称
                'description': str,   # 技能描述（来自 SKILL.md）
                'tools': List[BaseTool],  # 技能专属工具列表
                'prompts': Dict[str, str] # 技能专属提示词
            }
        """
        # 检查是否已加载
        if skill_name in self._loaded_skills:
            return self._loaded_skills[skill_name]
        
        skill_path = self.skills_dir / skill_name
        
        # 验证技能包是否存在
        if not skill_path.exists():
            raise ValueError(f"技能包 '{skill_name}' 不存在")
        
        if not (skill_path / "SKILL.md").exists():
            raise ValueError(f"技能包 '{skill_name}' 缺少 SKILL.md 文件")
        
        # 读取 SKILL.md 文件
        skill_description = (skill_path / "SKILL.md").read_text(encoding='utf-8')
        
        # 动态加载 tools.py
        tools = []
        try:
            tools_module = importlib.import_module(
                f"src.modules.agent.infrastructure.skills.{skill_name}.tools"
            )
            # 提取模块中所有的 BaseTool 实例
            for name in dir(tools_module):
                obj = getattr(tools_module, name)
                if isinstance(obj, BaseTool):
                    tools.append(obj)
        except ImportError:
            print(f"⚠️  技能包 '{skill_name}' 未定义 tools.py")
        
        # 动态加载 prompts.py
        prompts = {}
        try:
            prompts_module = importlib.import_module(
                f"src.modules.agent.infrastructure.skills.{skill_name}.prompts"
            )
            # 提取模块中所有的字符串变量
            for name in dir(prompts_module):
                if not name.startswith('_'):
                    obj = getattr(prompts_module, name)
                    if isinstance(obj, str):
                        prompts[name] = obj
        except ImportError:
            print(f"⚠️  技能包 '{skill_name}' 未定义 prompts.py")
        
        # 缓存已加载的技能
        skill_data = {
            'name': skill_name,
            'description': skill_description,
            'tools': tools,
            'prompts': prompts
        }
        
        self._loaded_skills[skill_name] = skill_data
        print(f"✅ 已加载技能：{skill_name} (包含 {len(tools)} 个工具)")
        
        return skill_data
    
    def create_load_skill_tool(self) -> BaseTool:
        """
        创建 load_skill 工具，供 Agent 按需调用
        
        Returns:
            LangChain 工具对象
        """
        @tool
        def load_skill(skill_name: str) -> str:
            """
            按需加载专业技能的 prompt 和上下文。
            
            当需要特定领域的帮助时使用此工具。
            参数 skill_name 必须是可用技能列表中的名称。
            
            返回的内容会被注入到后续对话上下文中。
            """
            try:
                skill_data = self.load_skill(skill_name)
                
                # 构建技能加载成功的响应
                response = f"已加载技能：{skill_data['name']}\n\n"
                response += f"描述:\n{skill_data['description']}\n\n"
                
                if skill_data['tools']:
                    tool_names = [t.name for t in skill_data['tools']]
                    response += f"可用工具：{', '.join(tool_names)}\n"
                
                return response
            except Exception as e:
                return f"加载技能失败：{e}"
        
        return load_skill
    
    def load_all_skills_eagerly(self) -> List[BaseTool]:
        """
        【可选】预加载所有技能并返回所有工具
        适用于启动时初始化场景
        
        Returns:
            所有技能的完整工具列表
        """
        all_tools = []
        for skill_name in self._available_skills:
            try:
                skill_data = self.load_skill(skill_name)
                all_tools.extend(skill_data['tools'])
            except Exception as e:
                print(f"❌ 加载技能失败 {skill_name}: {e}")
        
        return all_tools

# ==========================================
# 使用示例
# ==========================================

# 方式 1：按需加载（推荐）

load_skill_tool = loader.create_load_skill_tool()

# 将 load_skill_tool 注入到 Agent 的工具列表中
# agent = create_agent(

#         f"{loader.get_available_skills_list()}"
#     ),
# )

# 方式 2：启动时预加载所有工具

#### 8. LangGraph ReAct Agent 定义 (`src/modules/agent/infrastructure/graph/react_agent.py`)

from langchain.agents import create_agent
from langchain_core.language_models.chat_models import BaseChatModel
from ..tools.registry import AGENT_TOOLS
from ..workspace.prompts import AGENT_SYSTEM_PROMPT

def build_react_agent(llm: BaseChatModel, checkpointer):
    """
    构建并返回一个 LangChain 1.0 ReAct Agent 执行器
    使用新的 create_agent API（已取代 langgraph.prebuilt.create_react_agent）
    """
    # 动态组装 Prompt
    tool_desc = ", ".join([f"{t.name}: {t.description}" for t in AGENT_TOOLS])
    system_message = AGENT_SYSTEM_PROMPT.format(tool_descriptions=tool_desc)
    
    # 使用 LangChain 1.0 的新 API 创建 Agent
    # 注意：参数从 prompt 改为 system_prompt，支持中间件系统
    agent_executor = create_agent(
        model=llm,
        tools=AGENT_TOOLS,
        system_prompt=system_message,
        checkpointer=checkpointer, # 接入 PostgreSQL 持久化记忆
    )
    
    return agent_executor
```

#### 9. App Service 应用服务 (`src/modules/agent/application/services/agent_app_service.py`)

```python
import json
from typing import AsyncGenerator
from langchain_core.messages import HumanMessage

from ..dto.agent_chat_dto import AgentChatRequest
from ...infrastructure.adapters.thinking_adapter import ThinkingAdapter
from ...infrastructure.graph.react_agent import build_react_agent
from ...infrastructure.memory.postgres_saver import PostgresSaverManager
# 假设你现有的 providers 模块有获取模型实例的方法
# from src.modules.providers.application.services.chat_service import get_langchain_llm

class AgentAppService:
    def __init__(self, db_uri: str):
        # 初始化 PostgreSQL 连接管理器
        self.db_uri = db_uri
        self.checkpointer = None

    async def _get_checkpointer(self):
        """懒加载获取 PostgreSQL Checkpointer"""
        if self.checkpointer is None:
            self.checkpointer = await PostgresSaverManager.create_and_setup(self.db_uri)
        return self.checkpointer

    async def _prepare_agent(self, request: AgentChatRequest):
        """准备和初始化 Agent 执行器"""
        
        # 1. 获取 PostgreSQL Checkpointer
        checkpointer = await self._get_checkpointer()
        
        # 2. 调用现有的 providers 模块获取底层的 Langchain ChatModel 实例
        # 伪代码：llm = await get_langchain_llm(request.provider, request.model_name)
        from langchain_openai import ChatOpenAI # 仅作占位示意
        base_llm = ChatOpenAI(model=request.model_name) 
        
        # 3. 注入 Thinking 思考适配层（核心！）
        llm_with_thinking = ThinkingAdapter.apply_thinking(
            llm=base_llm,
            provider=request.provider,
            model_name=request.model_name,
            thinking=request.thinking
        )
        
        # 4. 构建 LangChain 1.0 ReAct Agent（传入 PostgreSQL Checkpointer）
        agent = build_react_agent(llm_with_thinking, checkpointer)
        return agent

    async def chat_sync(self, request: AgentChatRequest) -> str:
        """非流式对话"""
        agent = await self._prepare_agent(request)
        config = {"configurable": {"thread_id": request.session_id}}
        
        inputs = {"messages": [HumanMessage(content=request.message)]}
        response = await agent.ainvoke(inputs, config=config)
        
        # 提取最后一条 AI 回复
        return response["messages"][-1].content

    async def chat_stream(self, request: AgentChatRequest) -> AsyncGenerator[str, None]:
        """流式对话 (SSE 格式或 Chunk 格式)"""
        agent = await self._prepare_agent(request)
        config = {"configurable": {"thread_id": request.session_id}}
        inputs = {"messages": [HumanMessage(content=request.message)]}
        
        # 使用 LangChain 1.0 的 astream_events 来精细捕获流式输出（支持工具调用和模型思考过程输出）
        async for event in agent.astream_events(inputs, config=config, version="v2"):
            kind = event["event"]
            
            # 捕获大模型生成的文本块（含思考过程或最终回答）
            if kind == "on_chat_model_stream":
                chunk = event["data"]["chunk"]
                
                # 如果模型供应商返回了特殊的 reasoning_content (如 Deepseek)
                if hasattr(chunk, "additional_kwargs") and "reasoning_content" in chunk.additional_kwargs:
                    reasoning_text = chunk.additional_kwargs["reasoning_content"]
                    if reasoning_text:
                         yield f"data: {json.dumps({'type': 'thinking', 'content': reasoning_text})}\n\n"
                
                # 常规内容
                if chunk.content:
                    yield f"data: {json.dumps({'type': 'message', 'content': chunk.content})}\n\n"
            
            # 捕获工具调用开始
            elif kind == "on_tool_start":
                tool_name = event["name"]
                yield f"data: {json.dumps({'type': 'tool_call', 'content': f'正在使用工具：{tool_name} ...'})}\n\n"
                
        yield "data: [DONE]\n\n"
```

### 三、 集成与注册说明

1. **安装依赖**：
   
   在 `backend/pyproject.toml` 或 `requirements.txt` 中添加以下依赖：
   
   ```bash
   # LangChain 1.0 核心库
   pip install langchain>=1.0.0 langchain-core>=1.0.0
   
   # PostgreSQL 持久化支持
   pip install langgraph-checkpoint-postgres psycopg
   
   # 模型供应商 SDK（根据需要使用）
   pip install langchain-openai langchain-anthropic
   ```
   
   或者在 `pyproject.toml` 中：
   ```toml
   [project]
   dependencies = [
       # ... 其他依赖
       "langchain>=1.0.0",
       "langchain-core>=1.0.0",
       "langgraph-checkpoint-postgres>=0.0.1",
       "psycopg>=3.1.0",
   ]
   ```

2. **配置 PostgreSQL 连接**：
   
   在 `backend/.env` 或 `backend/config/settings.py` 中添加 PostgreSQL 连接字符串：
   
   ```python
   # PostgreSQL 连接配置
   LANGGRAPH_DB_URI = "postgresql://user:password@localhost:5432/my_database"
   # 或者使用环境变量
   # LANGGRAPH_DB_URI = os.getenv("LANGGRAPH_DB_URI")
   ```

3. **注册路由**：
   在现有的 `src/api/v1/router.py` 文件中，导入并注册新模块的路由：
   ```python
   from src.modules.agent.api.router import router as agent_router
   api_v1_router.include_router(agent_router, prefix="/agent")
   ```

4. **对接已有的 `providers` 模块**：
   在 `AgentAppService._prepare_agent` 方法中，你需要对接现有系统 `providers` 目录里的 `ChatService` 或 `Factories`。现有的 `providers` 既然已经封装好了，只需让它返回一个满足 `BaseChatModel`（Langchain 标准）的实例，然后交给 `ThinkingAdapter` 加工即可。

5. **Thinking 字段的灵活控制**：
   你的要求被完美实现在了 `ThinkingAdapter` 中。无论外部请求怎样传入 `thinking: true`，Adapter 都会将它翻译为当前 `provider` 对应的特殊参数（如改模型名为 `deepseek-reasoner`，或增加 `reasoning_effort="high"` 参数）。这使得前端和 API 调用方完全解耦。

### 五、 LangChain 1.0 迁移说明

本文档基于 **LangChain 1.0** 编写，与旧版本的主要区别如下：

#### 核心变化

1. **导入路径变更**：
   - ❌ 旧版：`from langgraph.prebuilt import create_react_agent`
   - ✅ 新版：`from langchain.agents import create_agent`

2. **参数名称变更**：
   - ❌ 旧版：`state_modifier=prompt`
   - ✅ 新版：`system_prompt=prompt`

3. **更强大的中间件系统**：
   - 旧版使用 Hook 进行前置/后置处理
   - 新版通过 **middleware（中间件）** 实现，提供更灵活的自定义能力

4. **工具描述增强**：
   - 建议在 `tool_desc` 中包含工具的名称和描述，帮助模型更好理解工具用途
   - 示例：`f"{t.name}: {t.description}"`

#### 核心特性继承

`create_agent` 依然构建在 **LangGraph** 之上，完整继承以下特性：
- ✅ 持久化存储（PostgreSQL Checkpointer）
- ✅ 检查点机制（Checkpointing）
- ✅ 流式传输（Streaming）
- ✅ 线程管理（Thread ID）
- ✅ 历史回溯（History Navigation）

#### 推荐实践

1. **使用 `astream_events`**：捕获 Agent 执行过程中的所有事件（工具调用、思考过程、最终输出）
2. **配置 `thread_id`**：通过 `config={"configurable": {"thread_id": "..."}}` 管理会话状态
3. **启用 PostgreSQL**：生产环境务必使用 `PostgresSaver` 而非内存存储
